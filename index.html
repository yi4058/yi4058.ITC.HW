  <script>

    (function() {

      const kiteContainer = document.getElementById('kiteContainer');

      const resetBtn = document.getElementById('resetBtn');

      const searchBtn = document.getElementById('searchBtn');

      const messageDiv = document.getElementById('message');

      const animals = ['🐶', '🐱', '🐼', '🦊', '🐸', '🐨', '🐧', '🐰', '🦝', '🐥'];

      const KITE_COUNT = 8;


      const questionBank = [

        { figure: '擬人', sentence: '頑皮的小雨滴在荷葉上盡情地跳舞。' },

        { figure: '擬物', sentence: '在教官嚴厲的注視下，他嚇得夾著尾巴逃跑了。' },

        { figure: '連續反復', sentence: '快跑！快跑！後面的追兵就要趕上來了。' },

        { figure: '反語', sentence: '你的動作可真快啊，讓我們足足等了半個小時！' },

        { figure: '反語', sentence: '我那時真是聰明得過份，總覺他說話不大漂亮，非自己插嘴不可。' },

        { figure: '呼告', sentence: '哥哥，你在國外一定要好好地照顧自己啊！' },

        { figure: '呼告', sentence: '大自然啊，你的創造力是如此的驚人！' },

        { figure: '呼告', sentence: '江河啊，請你把我的憂愁带走吧！' },

        { figure: '反問', sentence: '難道你不知道在課室裏是需要保持安靜的嗎？' },

        { figure: '反問', sentence: '難道你不知道沉迷網上遊戲是不健康的嗎？' },

        { figure: '設問', sentence: '學跆拳道有什麼好處？可以強身健體。' },

        { figure: '疑問', sentence: '這道題的正確解法是什麼？' },

      ];


      const TOTAL_QUESTIONS = questionBank.length;

      let usedIndices = [];

      let currentCenteredKite = null;


      function getRandomUnusedQuestion() {

        if (usedIndices.length >= TOTAL_QUESTIONS) {

          usedIndices = [];

          showMessage('🔄 題庫已循環，重新開始抽題', false);

        }

        const availableIndices = [];

        for (let i = 0; i < TOTAL_QUESTIONS; i++) {

          if (!usedIndices.includes(i)) availableIndices.push(i);

        }

        const randomIdx = availableIndices[Math.floor(Math.random() * availableIndices.length)];

        usedIndices.push(randomIdx);

        return { ...questionBank[randomIdx], index: randomIdx };

      }


      function showMessage(text, isError = false) {

        messageDiv.innerHTML = text;

        messageDiv.style.background = isError ? '#9b1c1c' : '#05445e';

        messageDiv.style.color = '#f0f9ff';

      }


      function createKites() {

        kiteContainer.innerHTML = '';

        for (let i = 0; i < KITE_COUNT; i++) {

          const kite = document.createElement('div');

          kite.className = 'kite';

          const left = 5 + Math.random() * 80;

          const top = 5 + Math.random() * 70;

          kite.style.left = left + '%';

          kite.style.top = top + '%';

          kite.dataset.originalLeft = left;

          kite.dataset.originalTop = top;

          const duration = 6 + Math.floor(Math.random() * 8);

          kite.style.animationDuration = duration + 's';

          kite.style.animationDelay = (Math.random() * 3) + 's';

          const animalEmoji = animals[i % animals.length];

          kite.setAttribute('data-animal', animalEmoji);

          const faceDiv = document.createElement('div');

          faceDiv.className = 'animal-face';

          faceDiv.textContent = animalEmoji;

          const lineDiv = document.createElement('div');

          lineDiv.className = 'kite-line';

          const bubble = document.createElement('div');

          bubble.className = 'explanation-bubble';

          kite.appendChild(faceDiv);

          kite.appendChild(lineDiv);

          kite.appendChild(bubble);

          kiteContainer.appendChild(kite);

        }

      }


      function adjustFontSizeByLength(element, text) {

        if (!element) return;

        element.style.fontSize = '';

        const len = text.length;

        const minLen = 5;

        const maxLen = 80;

        const minFont = 1.1;

        const maxFont = 2.6;

        const clampedLen = Math.min(maxLen, Math.max(minLen, len));

        const fontSize = maxFont - ((clampedLen - minLen) / (maxLen - minLen)) * (maxFont - minFont);

        element.style.fontSize = fontSize + 'rem';

      }


      function restoreKitePosition(kite) {

        if (!kite) return;

        kite.classList.remove('centered', 'stopped');

        if (kite.dataset.originalLeft && kite.dataset.originalTop) {

          kite.style.left = kite.dataset.originalLeft + '%';

          kite.style.top = kite.dataset.originalTop + '%';

        }

        const bubble = kite.querySelector('.explanation-bubble');

        if (bubble) {

          bubble.style.display = 'none';

          bubble.innerHTML = '';

          bubble.style.fontSize = '';

        }

        delete kite.dataset.currentFigure;

        delete kite.dataset.currentSentence;

      }


      function centerKite(kite) {

        if (currentCenteredKite && currentCenteredKite !== kite) {

          restoreKitePosition(currentCenteredKite);

        }

        if (!kite.dataset.originalLeft) {

          const left = parseFloat(kite.style.left) || 50;

          const top = parseFloat(kite.style.top) || 50;

          kite.dataset.originalLeft = left;

          kite.dataset.originalTop = top;

        }

        kite.classList.add('centered', 'stopped');

        const bubble = kite.querySelector('.explanation-bubble');

        if (bubble) bubble.style.display = '';

        currentCenteredKite = kite;

        document.body.classList.add('focus-mode');

      }


      function resetAllKites() {

        const kites = document.querySelectorAll('.kite');

        kites.forEach(kite => restoreKitePosition(kite));

        currentCenteredKite = null;

        document.body.classList.remove('focus-mode');

        showMessage('✨ 風箏恢復飄動，可繼續點擊抽取新句子');

      }


      function showCurrentFigure() {

        if (!currentCenteredKite) {

          showMessage('❓ 沒有正在觀看的風箏，請先點擊一個風箏', true);

          return;

        }

        const figure = currentCenteredKite.dataset.currentFigure;

        const sentence = currentCenteredKite.dataset.currentSentence || '';

        if (!figure) {

          showMessage('⚠️ 這個風箏還沒有句子，請重新點擊', true);

          return;

        }

        messageDiv.innerHTML = `🔍 修辭手法：<span class="figure-badge">${figure}</span><br><span style="font-size:1rem;">「${sentence}」</span>`;

        messageDiv.style.background = '#1e5f7a';

        messageDiv.style.color = '#fef9e0';

      }


      function onKiteClick(e) {

        const kite = e.target.closest('.kite');

        if (!kite) return;

        const isSameKite = (currentCenteredKite === kite);

        if (!isSameKite) {

          if (currentCenteredKite) restoreKitePosition(currentCenteredKite);

          centerKite(kite);

        } else {

          if (!kite.classList.contains('centered')) centerKite(kite);

        }

        const bubble = kite.querySelector('.explanation-bubble');

        if (!bubble) return;

        const { figure, sentence } = getRandomUnusedQuestion();

        bubble.innerHTML = sentence;

        adjustFontSizeByLength(bubble, sentence);

        kite.dataset.currentFigure = figure;

        kite.dataset.currentSentence = sentence;

        showMessage(`✨ 句子已送達風箏。按🔍查看修辭`, false);

      }


      kiteContainer.addEventListener('click', onKiteClick);

      resetBtn.addEventListener('click', resetAllKites);

      searchBtn.addEventListener('click', showCurrentFigure);


      createKites();

      showMessage(`🎏 點擊風箏抽取句子（共 ${TOTAL_QUESTIONS} 種修辭，不重複）`);

    })();
